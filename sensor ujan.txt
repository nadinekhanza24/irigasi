#define BLYNK_TEMPLATE_NAME "Rain dan LDR"
#define BLYNK_AUTH_TOKEN    "ljOvSS75rjcFe4nN3nNnr3lswKhfAfTr"

#include <ESP8266WiFi.h>
#include <BlynkSimpleEsp8266.h>
#include <Servo.h>

char auth[] = BLYNK_AUTH_TOKEN;
char ssid[] = "POCO";     
char pass[] = "embedded"; 

const int pinHujan  = D5;
const int pinLDR    = A0;
const int pinServo  = D4;
const int pinBuzzer = D1;

Servo myservo;
BlynkTimer timer;

const int batasTerang  = 200; 
const int batasMendung = 350; 

bool statusBuzzer = false; 

void kirimDataSensor() {
  int statusHujan = digitalRead(pinHujan); // LOW = Hujan, HIGH = Tidak Hujan
  int nilaiLDR = analogRead(pinLDR);

  // =============================================
  // KONDISI 1: HUJAN (Siang atau Malam) → Buzzer BUNYI
  // =============================================
  if (statusHujan == LOW) {
    myservo.write(0); // Jemuran MASUK (0 derajat)

    // Efek bip-bip setiap 1 detik
    statusBuzzer = !statusBuzzer;
    if (statusBuzzer) {
      tone(pinBuzzer, 1000); // BUNYI 1000 Hz
    } else {
      noTone(pinBuzzer);     // DIAM (efek bip)
    }

    // Hujan Siang: LDR Terang
    if (nilaiLDR < batasTerang) {
      Blynk.virtualWrite(V1, "Hujan Siang (Jemuran Masuk)");
    }
    // Hujan Malam: LDR Gelap
    else {
      Blynk.virtualWrite(V1, "Hujan Malam (Jemuran Masuk)");
    }
  }

  // =============================================
  // KONDISI 2: TIDAK HUJAN → Buzzer MATI
  // =============================================
  else {
    noTone(pinBuzzer); // Buzzer selalu MATI jika tidak hujan
    statusBuzzer = false;

    // A. Cerah (Terang) → Jemuran KELUAR
    if (nilaiLDR < batasTerang) {
      myservo.write(90); // Jemuran KELUAR (90 derajat sesuai permintaan)
      Blynk.virtualWrite(V1, "Cerah (Jemuran Keluar)");
    }
    // B. Mendung (Agak Gelap) → Jemuran TETAP KELUAR
    else if (nilaiLDR >= batasTerang && nilaiLDR <= batasMendung) {
      myservo.write(90); // Jemuran TETAP KELUAR (90 derajat sesuai permintaan)
      Blynk.virtualWrite(V1, "Mendung (Jemuran Keluar)");
    }
    // C. Malam Tidak Hujan (Gelap) → Jemuran MASUK
    else {
      myservo.write(0); // Jemuran MASUK (0 derajat)
      Blynk.virtualWrite(V1, "Malam (Jemuran Masuk)");
    }
  }

  // Kirim nilai LDR ke Gauge Blynk (V2)
  Blynk.virtualWrite(V2, nilaiLDR);

  // Debug Serial Monitor
  Serial.print("Hujan: "); Serial.print(statusHujan);
  Serial.print(" | LDR: "); Serial.print(nilaiLDR);
  Serial.print(" | Buzzer: "); Serial.println(statusBuzzer);
}

void setup() {
  Serial.begin(115200);

  // --- PERBAIKAN DI SINI ---
  // Menggunakan INPUT_PULLUP agar pin D5 stabil menerima logika HIGH saat kering
  pinMode(pinHujan, INPUT_PULLUP);
  
  pinMode(pinBuzzer, OUTPUT);
  noTone(pinBuzzer); // Buzzer MATI di awal

  myservo.attach(pinServo);
  myservo.write(0); // Posisi awal jemuran MASUK

  Blynk.begin(auth, ssid, pass);
  timer.setInterval(1000L, kirimDataSensor);
}

void loop() {
  Blynk.run();
  timer.run();
}
